python scripts/09_severity_weighted_behavior_vectors/02_over/01_extract_over_behavior_vectors.py --selected-method severity_weighted --severity-beta 1.0

CUDA_VISIBLE_DEVICES=7 python scripts/09_severity_weighted_behavior_vectors/02_over/02_run_over_behavior_steering.py --split val --layer 32 --alpha 0.5

python scripts/09_severity_weighted_behavior_vectors/02_over/04_apply_over_conditional_test.py --layer 32 --alpha 0.5 --threshold SELECTED_VAL_THRESHOLD
